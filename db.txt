id INT PRIMARY KEY AUTO_INCREMENT,
marca VARCHAR(50) NOT NULL,
modello VARCHAR(50) NOT NULL,
anno_produzione INT NOT NULL,
chilometraggio INT NOT NULL,
prezzo DECIMAL(10, 2) NOT NULL,
colore VARCHAR(30),
carburante ENUM('Benzina', 'Diesel', 'GPL', 'Elettrico', 'Ibrido') NOT NULL,
cambio ENUM('Manuale', 'Automatico') NOT NULL,
potenza_cv INT, 
cilindrata INT,
numero_porte INT,
stato_veicolo VARCHAR(100),
proprietari_precedenti INT,
ultima_revisione DATE,
optional TEXT,
descrizione TEXT,
disponibilita BOOLEAN DEFAULT TRUE,
data_inserimento DATE DEFAULT CURRENT_DATE,
immagine_principale VARCHAR(255),
concessionario_id INT,
